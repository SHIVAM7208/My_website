Selenium WebDriver Cheat Sheet

WebDriver Setup

Firefox:  WebDriver driver = new FirefoxDriver();
                              
                                 Chrome:  WebDriver driver = new ChromeDriver();
                              
                                 Safari:  WebDriver driver = new SafariDriver();

Locators

                                      By CSS Selector:
                                      driver.findElement(By.CssSelector("#rightbar > .menu > li:nth-of-type(2) > h4"));
                                      
                                      By ID:
                                      driver.findElement(By.id("q")).sendKeys("Selenium 3");
                                      
                                      By Xpath:
                                      driver.findElement(By.xpath("//input[@id='q']")).sendKeys("Selenium 3");
                                      
                                      By Link Text:
                                      driver.findElement(By.linkText("NextPage")).click();

Handling Alerts

Accept Alert:
                                              driver.switchTo().alert().accept();
                                              
                                              Dismiss Alert:
                                              driver.switchTo().alert().dismiss();
                                              
                                              Send Keys to Alert:
                                              driver.switchTo().alert().sendKeys("Text");

Handling Cookies

Add Cookie:
                                              driver.manage().addCookie(new Cookie("foo", "bar"));
                                              
                                              Get Cookie:
                                              driver.manage().getCookieNamed("foo");
                                              
                                              Delete Cookie:
                                              driver.manage().deleteCookieNamed("foo");

Screenshots

                                        TakesScreenshot screenShot = (TakesScreenshot)driver;

                                        FileHandler.copy(screenShot.getScreenshotAs(OutputType.FILE), new File("path/to/destination/folder/screenshot.png"));

Navigation

                                              Navigate Back:
                                              driver.navigate().back();
                                              
                                              Navigate Forward:
                                              driver.navigate().forward();
                                              
                                              Refresh Page:
                                              driver.navigate().refresh();

Window and Frame Handling

                                              Switch to Frame by Index:
                                              driver.switchTo().frame(0);
                                              
                                              Switch to Frame by Name:
                                              driver.switchTo().frame("frameName");
                                              
                                              Switch to Main Content:
                                              driver.switchTo().defaultContent();

WebDriverWait

 WebDriverWait wait = new WebDriverWait(driver, 20);
                                       wait.until(ExpectedConditions.textToBePresentInElementLocated(By.xpath("//div[@id="Object_ID"]"),"Details"));

Fluent Wait

Wait wait = new FluentWait(driver)
                                      .withTimeout(Duration.ofSeconds(20))
                                      .pollingEvery(Duration.ofSeconds(2))
                                      .ignoring(NoSuchElementException.class);

Actions Class for Mouse and Keyboard Events

Actions builder = new Actions(driver);
                                      builder.moveToElement("login-textbox").click().keyDown(Keys.SHIFT)
                                      .sendKeys("hello").keyUp(Keys.SHIFT).doubleClick().contextClick().build().perform();

Execute JavaScript

JavascriptExecutor js = (JavascriptExecutor) driver;
                                      js.executeScript("alert('Hello World');");

Switch Windows

String mainWindowHandle = driver.getWindowHandle();
                                      Set<String> allWindowHandles = driver.getWindowHandles();
                                      for(String handle : allWindowHandles) {
                                              driver.switchTo().window(handle);
                                      }
